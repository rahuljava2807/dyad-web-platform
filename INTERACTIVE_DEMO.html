// YaviClient.tsx - Yavi.ai Integration Component
import React, { useState, useEffect } from 'react'
import { YaviDataClient } from '@yavi/data-sdk'

interface YaviClientProps {
  apiKey: string
  onProcessingComplete?: (result: any) => void
}

export const YaviClient: React.FC<YaviClientProps> = ({
  apiKey,
  onProcessingComplete
}) => {
  const [client, setClient] = useState<YaviDataClient | null>(null)
  const [isProcessing, setIsProcessing] = useState(false)
  const [documents, setDocuments] = useState<any[]>([])

  useEffect(() => {
    const yaviClient = new YaviDataClient({
      apiKey: apiKey
    })
    setClient(yaviClient)
  }, [apiKey])

  const processDocument = async (document: File) => {
    if (!client) return

    setIsProcessing(true)
    try {
      const result = await client.processDocument({
        document: document,
        workflow: 'intelligent-extraction'
      })

      setDocuments(prev => [...prev, result])
      onProcessingComplete?.(result)
    } catch (error) {
      console.error('Document processing failed:', error)
    } finally {
      setIsProcessing(false)
    }
  }

  return (
    <div className="yavi-client">
      <h3>Document Intelligence Hub</h3>
      {/* File upload and processing UI */}
      <div className="upload-zone">
        <input
          type="file"
          onChange={(e) => {
            const file = e.target.files?.[0]
            if (file) processDocument(file)
          }}
        />
      </div>
    </div>
  )
}

export default YaviClient

// Document Processing with Yavi.ai
export class DocumentIntelligenceService {
  async processContract(file: File, options = {}) {
    try {
      // Upload document to Yavi
      const uploadResult = await yaviClient.documents.upload({
        file: file,
        type: 'contract',
        metadata: {
          source: 'dyad-platform',
          uploadedBy: options.userId
        }
      })

      // Process with intelligent extraction
      const analysis = await yaviClient.intelligence.analyze({
        documentId: uploadResult.documentId,
        workflow: 'contract-analysis',
        features: [
          'clause-extraction',
          'risk-assessment',
          'compliance-check',
          'entity-recognition'
        ],
        options: {
          confidenceThreshold: 0.85,
          generateSummary: true,
          extractTables: true
        }
      })

      // Return structured results
      return {
        documentId: analysis.documentId,
        status: analysis.status,
        extractedData: analysis.results,
        confidence: analysis.confidence,
        processingTime: analysis.processingTimeMs,
        costEstimate: analysis.usage
      }
    } catch (error) {
      throw new Error(`Document processing failed: ${error.message}`)
    }
  }
}

// Secure API calls with retry logic
class YaviSecureClient {
  constructor(config) {
    this.apiKey = config.apiKey
    this.baseURL = config.baseURL
    this.timeout = config.timeout || 30000
  }

  async authenticatedRequest(endpoint, data) {
    const headers = {
      'Authorization': `Bearer ${this.apiKey}`,
      'Content-Type': 'application/json',
      'X-Client-Version': 'dyad-v1.0.0'
    }

    return await this.retryRequest(endpoint, {
      method: 'POST',
      headers,
      body: JSON.stringify(data)
    })
  }
}
